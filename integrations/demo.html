üîó Integration Examples

Web Applications

// React Integration import { BuddyClient } from './buddy-sdk'; function ChatComponent() { const [buddy] = useState(new BuddyClient('http://localhost:8081')); const [messages, setMessages] = useState([]); const sendMessage = async (text) => { const response = await buddy.chat(text); setMessages(prev => [...prev, response]); }; return <ChatInterface onSend={sendMessage} messages={messages} />; }

Discord Bot

# Discord Bot with BUDDY import discord from buddy_sdk import BuddyClient buddy = BuddyClient("http://localhost:8081") @bot.event async def on_message(message): if message.author == bot.user: return response = await buddy.chat_async(message.content) await message.channel.send(response.message)

Express.js API

// Express API with BUDDY const express = require('express'); const { BuddyClient } = require('./buddy-sdk'); const app = express(); const buddy = new BuddyClient('http://localhost:8081'); app.post('/chat', async (req, res) => { try { const response = await buddy.chat(req.body.message); res.json(response); } catch (error) { res.status(500).json({ error: error.message }); } });
